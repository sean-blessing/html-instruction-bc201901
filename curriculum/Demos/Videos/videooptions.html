<!-- JavaScript to read the Query String for options -->
<script>
 var url = window.location.href;
 var parts = url.split('?');
 if (parts.length==1)
  { document.getElementById('videoarea').innerHTML = '<b>Select one or more options</b>'; }
  else
  {
    var options = parts[1].split('&');
    var text = "";
    for (var i=0; i<options.length; i++)
    {
      text += unescape(options[i].split('=')[1]) + ' ';
    }
    document.getElementById('videoarea').innerHTML = ' \
      &lt;video id="video" width="320" height=178   \
      src="snow320x178.mp4" type="video/mp4"  ' + text +' &gt;<br>  \
      \
      <video id="video" width="320" height=178 ' + text +' > \
        <source src="snow320x178.mp4" type="video/mp4"> \
        Your browser does not support the video tag.  \
      </video>'
  }
</script>
