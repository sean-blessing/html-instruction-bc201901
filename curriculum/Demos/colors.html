Or use sliders: <br>
  Red: <input id=redslider type=range max=255 min=0 value=255 oninput="setcolor2()" onchange="setcolor2()">
  Green: <input id=greenslider type=range max=255 min=0 value=0 oninput="setcolor2()" onchange="setcolor2()">
  Blue: <input id=blueslider type=range max=255 min=0 value=0 oninput="setcolor2()" onchange="setcolor2()">

<script>
// button click
function setcolor(frombutton)
{

  var vred    = ("00"+document.getElementById("red").value).slice(-2);
  var vgreen  = ("00"+document.getElementById("green").value).slice(-2);
  var vblue   = ("00"+document.getElementById("blue").value).slice(-2);
  var color  = "#"+ vred + "" + vgreen + "" + vblue;
document.getElementById("out").innerText = color
  document.getElementById("colorbox").setAttribute("style", 
   "float:left;border:1px solid;width:100px;height:100px;background-color:" + color);  

  if (frombutton)
  {
    document.getElementById("redslider").value = parseInt(vred, 16);
    document.getElementById("greenslider").value = parseInt(vgreen, 16);
    document.getElementById("blueslider").value = parseInt(vblue, 16);
  }
}

//sliders
function setcolor2()
{
  //event.stopImmediatePropagation()
  var vred    = document.getElementById("redslider").value;
  var vgreen  = document.getElementById("greenslider").value;
  var vblue   = document.getElementById("blueslider").value;
  document.getElementById("red").value = ("00" + parseInt(vred).toString(16)).slice(-2);
  document.getElementById("green").value = ("00" + parseInt(vgreen).toString(16)).slice(-2);
  document.getElementById("blue").value = ("00" + parseInt(vblue).toString(16)).slice(-2);

  setcolor(false);
}

</script>
